코드 1
fun solution(numbers: IntArray): String {
    var answer = ""
    numbers.sortedWith(
        Comparator { o1, o2 ->
            "$o2$o1".compareTo("$o1$o2")
        }
    ).forEach { 
        answer += it 
    }

    if(answer[0] == '0') {
        answer = "0"
    }
    return answer
}
코드 2 : 다른 사람의 코드
fun solution(numbers: IntArray): String {
    var answer = ""        
    numbers.sortedWith(Comparator({num1: Int, num2: Int -> "$num2$num1".compareTo("$num1$num2")})).forEach { answer += it }
    if ("(0*)".toRegex().replace(answer, "").isEmpty()) {
        answer = "0"
    }
    return answer
}
